Employee Pseudocodes
LOGIN

START
1. Display login screen
2. Prompt user to enter email and password
3. Set Email & Password varialbe = email & password
4. If email & password == email & password in database
	Display login successful
   else 
	Display invalid credentials
	Prompt user to enter new email & password
   endif
END

LOGOUT

START
1. Display logout button
2. Read user input
3. If user selects logout
	Prompt user to confirm logout
	If user confirms logout
		Stop user session
	Endif
   Endif
		

PASSWORD RESET

START
1. Display 'password reset button'
2. If user selects 'password reset button'
	Initialise token
	Send token to user's email
	Prompt user to enter token sent
	If token == initialised token
		Prompt user to enter old password & new password & confirm new password
		If password == old password
			Set password = new password
   		else 
			display invalid password
			prompt user to reenter old password
		endif
   	else
		display invalid token
		prompt user to reenter token
   	endif
   Endif
END

SEND MESSAGES

START
1. Prompt user to select receipient
2. Prompt user to enter message
3. Set sender id & receipient id & message variables = sender's id & receipient id & message
4. If message is sent
	Display sent
   endif
END

THREAD OF USERS

START
1. Set id & name = id & name in database
2. repeat while user exits in database
	display user's name
   end repeat
END

PIN USER

START
1. Prompt user to select user to pin
2. Set pinned_id = user's id
3. Repeat while pinned_ids exist in database
	Display pinned users
   End repeat
END

FREQUENTLY CONTACTED USERS

START
1. Foreach reciepient, set message_counter
2. If reciepient message_counter > 0
	Sort most contacted users in descending orders
   End if
3. Repeat while most_contacted_users <=5 
	Display most_contacted_users in frequently contacted section
   End repeat
END

DISPLAYING LIST OF PERSONNEL USER CAN CHAT

START
1. Set id & name = id & name in database
2. If receipients employee_category_id = user's category_id
	repeat while user exits in database
		display user's name
   	end repeat
   endif
END

SET STATUS FROM LIST OF PRESET STATUSES

START
1. Prompt user to set status
2. Repeat while preset statuses exist in database
	Display list of status
3. Read user's input from keyboard
4. Set user status_id = selected status_id
5. Display user status
END

DISPLAY AND REPLY NOTIFICATION OF RECEIVED MESSAGES 

START
1. Read database for incomming message
2. If new message exists
	Display new message & sender name
	Prompt user to enter message
	Set sender id & receipient id & message variables = sender's id & receipient id & message
		If message is sent
			Display sent
   		endif
   endif
END

SCHEDULE TASK

START
1. Prompt user to schedule task
2. Display task schedular interface
3. Prompt user to enter task title & body
4. Read user's input from keyboard
5. Set task title & body variable & due date/time = task title & body & due date/time
END

EDIT TASK

START
1. Repeat while task exist in database
	Display task
   End repeat
2. Prompt user to select task to edit
3. Display selected task details
4. Read user inputs from keyboard
5. Set task title & body variable & due date/time = new task title & body & new date/time
END

DELETE TASK

START
1. Repeat while task exist in database
	Display task
   End repeat
2. Prompt user to select task to delete
3. Read user's input
4. Prompt user to confirm deletion
5. Read user's input from keyboard
6. If user confirms deletion
	Remove task record from database
   endif
END

SEE LIST OF COMPLETED TASK

START
1. Repeat while task exist in database
	If taskComplete = true
		Display task
	Endif
   End repeat
END

SEE LIST OF PENDING TASK
START
1. Repeat while task exist in database
	If taskComplete = false
		Display task
	Endif
   End repeat
END
